#
def mod(y):
    if len(y)==1:
            return y[0]
    numDic = {} #
    for i in y:
        if i in numDic.keys():
            numDic[i] += 1 #label+1
            if numDic.get(i)>=(len(y)+1)/2: #labellabel
                    return i
        else:
            numDic[i] = 1
    maxkey = 0
    maxvalue = 0
    for key, value in numDic.items():
        if value > maxvalue: maxkey = key
    return maxkey

#
def predict_(indices):
    pred = []
    for indice in indices:
        p_labels = train_y[indice] #predictlabel
        p = mod(p_labels) #
        pred.append(p)
    return pred

accuracy
def acc(p, y):
    count = 0
    for i, pred in enumerate(p):
        if p[i] == y[i]: count += 1
    return count / len(p)

#distance array, 1/16 * 256 = 16weight array
from copy import deepcopy
from math import exp

def kweight(dis, k = 16):
    weights = deepcopy(dis)
    for i, dist in enumerate(dis):
        weights[i] = exp(-dist*dist/(k*k))
    return weights

def predict3_(dis, indices, kk):
    pred = []
    for i, indice in enumerate(indices):
        p_labels = train_y[indice]
        w = kweight(dis[i], kk)
        uniq = list(set(p_labels))
        ww = []
        for label in uniq:
            wcount = 0
            for i, p in enumerate(p_labels):
                if p == label: wcount += w[i]
            ww.append(wcount)
        p = uniq[np.argmax(ww)]
        pred.append(p)
        
    return pred
